## 函数防抖

```js
function debounce(fn: any, delay: any = 1000, immediate: boolean = false) {
	let time: any = null;
    let isImmediateInvoke: boolean = false;
    console.log('-----------函数防抖----------');
    function _debounce(args: any) {
      if (time !== null) {
        clearTimeout(time);
      }
      if (!isImmediateInvoke && immediate) {
        console.log('函数先执行,再计时');
        fn(args);
        isImmediateInvoke = true;
      }
      time = setTimeout(() => {
        console.log('函数执行');
        fn(args);
        isImmediateInvoke = false;
      }, delay);
    }
    return _debounce;
  }
```


## 函数节流

```js

// 定时器方案

  function throttle(fn: Function, wait: number = 1000) {
    let timer: any = null;
	console.log('-----------函数节流----------');
    return function () {
      let args = arguments;
      if (!timer) {
        timer = setTimeout(function () {
		console.log('-----------函数执行----------');
          fn(args);
          timer = null;
        }, wait);
      }
    };
  }


```


## 使用

```js
// 防抖
 window.addEventListener(

    'mousemove',

    debounce(() => {

      console.log('鼠标移动');

    }, 3000),

  );

// 节流
 window.addEventListener(

    'mousemove',

    throttle(() => {

      console.log('鼠标移动');

    }, 3000),

  );
```
